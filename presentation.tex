\begin{frame}{1. Introduction \& Motivation}
\begin{columns}[T,onlytextwidth]
\column{0.58\textwidth}
\textbf{Electrified Autonomous Flexible Transit (E-AFT)}
\begin{itemize}\itemsep3pt
  \item Small batteries $\Rightarrow$ routing is charging-sensitive.
  \item Most prior work assumes a \alert{constant charging rate} (Jiang et al., 2026).
  \item Real charging process is \alert{tapering}: fast -> slow.
  % Charging tapering is the automatic reduction of EV charging speed as the battery nears full capacity
\end{itemize} \\[25pt]

\textbf{Research question}
\begin{block}{}
\emph{How does the scheduling (routing, charging) of E-AFT change under different charging assumptions (constant charging rate vs. nonlinear charging rate)?}
\end{block}

\begin{frame}{2. Problem Description}
\begin{columns}[T,onlytextwidth]
\column{0.62\textwidth}
\textbf{A static version of the trip routing and charging scheduling problem}
\begin{itemize}\itemsep2pt
  \item Pre-booked passenger requests: origin, destination, pickup time window, \# passengers.
  \item Each trip: a sequence of pickup/drop-off stops, subject to capacity and time windows.
  \item Between consecutive trips: arrive at a charging station.
  \item Limited battery $\Rightarrow$ routing (sequence of stops), timing, and charging are \emph{tightly coupled}.
\end{itemize}

% ================= MODEL OVERVIEW =================
\begin{frame}{3. Modeling Approach}
\textbf{Joint optimized decisions under two charging profiles}
\begin{itemize}\itemsep1pt
  \item Which requests are served/dropped out, by which trip.
  \item The order of arriving at pickup/drop-off stops inside each trip.
  % \item[$\bullet$] Initial dispatch station; inter-trip station choice.
  \item Charging duration \& acquired energy at each station visit.
\end{itemize}
\vspace{5pt}
\textbf{Objective} \\[2pt]
% \vspace{-4pt}
% \begin{align*}
% Z = \sum_q C_m y_q
%   - \sum_q C_z(1\!-\!y_q)
%   - \sum_k C_v\,u_k
%   - \sum_{k,r} C_t\,u_{kr}
%   - C_d \!\!\sum_{k,i,j,r}\!\! \tau_{ij} x_{kijr}
%   - \sum_{k,s,r} C_{e,s}\, E_{ksr}.
% \end{align*}
Maximize operating profit: revenue - penalty for not serving requests - bus operational cost - charging cost

\vspace{5pt}
\textbf{Constraints}
\begin{itemize}\itemsep1pt
  \item Request-to-trip assignment, bus-to-trip assignment, etc.
  % , trip activation, subsequent-trip ordering.
  % \item Arc-flow routing with in/out degree, first/last node indicators.
  \item Flow conservation
  \item Time-window constraint
  \item Energy constraint, inter-trip charging constraint
  % \item Passenger load propagation bounded by $C_b$.
  % \item Battery propagation within trips; inter-trip transitions via station.
  % \item Charging coupling $E_{ksr} = \mathrm{PWL}_s(\theta_{ksr})$ with station-off indicators.
  % \item Symmetry breaking across identical buses.
\end{itemize}
\end{frame}

% ================= CHARGING PWL =================
\begin{frame}{3. Modeling Approach}
\begin{columns}[T,onlytextwidth]
\column{0.55\textwidth}
\textbf{From bilinear $E = p\,\theta$ to PWL}
\begin{itemize}\itemsep2pt
  \item $p_s$: charging power (kW), $\theta$: charging duration, $E$: charged energy
  \item $E$ is simplified as a piecewise linear curve.
  \item Gurobi \texttt{addGenConstrPWL} enforces $E = f_s(\theta)$.
  % \item Indicator: $z_{ksr}=0 \Rightarrow \theta_{ksr}=E_{ksr}=0$.
\end{itemize}

\vspace{4pt}
\textbf{Linear curve}:\\[-20pt]
\[ (0,0),(4,4.8),(8,9.6),(12,14.4),(16,19.2) \]

\textbf{Nonlinear curve} (tapering):\\[-8pt]
\[ \begin{array}{l}
(0,0),(2.5,5.2),(5,9.6),(7.5,12.4),\\
(11.5,14.8),(18.5,16.8),(32,19.2)
\end{array} \]

\begin{frame}{4. Experimental Design}
\textbf{Setup.} 1 bus, 3 trip slots, 7 requests.\\
\textbf{Parameters.} Battery 36 kWh, min 4 kWh, initial 22 kWh, bus capacity 3 pax, penalty \$18/unserved, travel cost \$1.1/minute, charge cost \$0.55/kWh  
\vspace{6pt}
\begin{block}{Direct comparison: three scenarios (differ in pickup windows)}
\begin{tabular}{@{}p{0.19\textwidth}p{0.75\textwidth}@{}}
\textbf{Baseline}       & all 7 windows are relatively wide ($\sim$55 min)\\
\textbf{Partial recharge} & time windows of \texttt{r3,r4} are set to $\sim$10-min; a moderate recharge between trips $\Rightarrow$ partial recharge \\
\textbf{Deep recharge}  & time windows of \texttt{r3,r4,r5} are set closer; the third trip needs a substantial inter-trip recharge $\Rightarrow$ deep recharge
\end{tabular}
\end{block}

% \vspace{4pt}
\textbf{Cross-evaluation}
\begin{itemize}\itemsep1pt
  \item Solve for the solution under linear charging
  \item Evaluate the solved plan (route, station, request-trip assignment) under nonlinear charging 
  \item Re-optimize under nonlinear charging
\end{itemize}

\textbf{Multi-bus scenario}: re-run deep recharge scenario with 2 buses, 2 trip slots/bus, 8 requests.
\end{frame}

\end{block}

% ================= TAKEAWAY =================
\begin{frame}{5. Discussions \& Conclusions - Findings}
% \begin{block}{Operational implication}
% \end{block}

% \vspace{4pt}
% \textbf{What the experiments showed}
\begin{itemize}\itemsep2pt
  % \item A one-stage integrated MIP handles multi-station charging, station-specific speed \& price, dispatch-station decisions, and a tapering charging profile -- without the bilinear $E = p\,\theta$ term, by absorbing $p$ into a station-specific time-axis scaling of a shared PWL.
  \item Baseline charges \emph{longer} under nonlinear (41.27 vs 24.57 min for 35.95 kWh) because the demand reaches the slow tail of the nonlinear charging curve. \textbf{Tapering charging is not uniformly faster}.
  \item \textbf{Single-bus, $|K|=1$, $|R|=3$, $|Q|=7$, deep recharge}: Linear obj 165.51 (5 served) vs nonlinear 109.07 (4 served): \alert{\textbf{gap 56.44, and linear's plan is infeasible under nonlinear assumption}}.
  \item \textbf{Multi-bus, $|K|=2$, $|R|=2$, $|Q|=8$, deep recharge}: Similar finding: linear 300.22 vs nonlinear 293.98, \alert{\textbf{linear's plan is still infeasible under nonlinear assumption}}.
  \item The linear charging assumption is safe when there is enough time between requests and \textbf{misleading when requests' pickup times are close}. In this case, it produces a plan that the real battery with tapering charging cannot execute.
  % \item \textbf{Three buses, same instance.} The asymmetry collapses --- capacity is too loose for the deep-recharge regime to bind.
\end{itemize}
\end{frame}

% \begin{frame}{Backup --- Anticipated Questions}
% \small
% \begin{block}{Q1. Why not keep the bilinear $E = p\,\theta$?}
% Our implementation factors station-specific $p_s$ into the \emph{time-axis scaling} of a shared PWL curve. Gurobi's \texttt{addGenConstrPWL} then enforces $E = f_s(\theta)$ exactly on the approximation. This keeps the model MIP (not MINLP) while preserving the key nonlinear effect --- the tapering shape of the curve.
% \end{block}

% \begin{block}{Q2. Is the PWL approximation tight?}
% The taper uses 6 segments on $[0,32]$; slopes step from 2.08 down to 0.18 kWh/min, giving a piecewise upper envelope of a typical Li-ion SoC-vs-time curve. Finer grids are cheap (only one additional breakpoint per segment), but the course-size instance already terminates at MIP gap 0.
% \end{block}

% \begin{block}{Q3. Why does the asymmetry only show up in deep recharge?}
% Deep recharge is the only scenario where the bus has to acquire enough energy in one inter-trip gap to reach the slow tail of the taper. In baseline and partial recharge the recharge per gap stays inside the early segment, where the constant-rate and tapering curves disagree only on time, not on whether the energy is acquired at all. The pathology is therefore specific to the regime where capacity is tight enough to push a single recharge into the tail.
% \end{block}
% \end{frame}

% \begin{frame}{Backup --- Anticipated Questions (cont.)}
% \small
% \begin{block}{Q4. Why pick the hub station when it charges pricier?}
% Hub costs \$0.64/kWh vs.\ depot \$0.48/kWh, but its time-scale factor is 0.82 (vs.\ depot's 1.08). In the partial-recharge regime the binding constraint is \emph{time} (the squeezed \texttt{r3} pickup window), not \emph{cost}, so the solver pays for speed.
% \end{block}

% \begin{block}{Q5. How do we know the cross-evaluation is not picking the wrong plan?}
% The discrete binaries ($u_{kr},\,x_{kijr},\,l^{\text{first/last}},\,z_{ksr}$) are extracted from the source solve and re-imposed as equality bounds in the target solve. Continuous variables (arrivals, battery levels, charge time / energy) are still free --- only the routing \& station choice is fixed.
% \end{block}
